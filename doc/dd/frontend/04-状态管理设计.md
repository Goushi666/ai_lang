# Web 端设计 · 状态管理设计

## 1. 传感器数据 (`store/modules/sensor.js`)

```javascript
import { defineStore } from 'pinia'
import { sensorApi } from '@/api/sensor'

export const useSensorStore = defineStore('sensor', {
  state: () => ({
    latestData: null,
    historyData: [],
    loading: false
  }),

  actions: {
    async fetchLatest() {
      this.loading = true
      try {
        this.latestData = await sensorApi.getLatest()
      } finally {
        this.loading = false
      }
    },

    async fetchHistory(params) {
      this.loading = true
      try {
        this.historyData = await sensorApi.getHistory(params)
      } finally {
        this.loading = false
      }
    },

    updateRealtimeData(data) {
      this.latestData = data
    }
  }
})
```

## 2. 车辆控制 (`store/modules/vehicle.js`)

```javascript
import { defineStore } from 'pinia'
import { vehicleApi } from '@/api/vehicle'

export const useVehicleStore = defineStore('vehicle', {
  state: () => ({
    status: {
      mode: 'manual',
      speed: 0,
      connected: false
    },
    controlEnabled: false
  }),

  actions: {
    async sendCommand(command) {
      await vehicleApi.sendControl(command)
    },

    updateStatus(status) {
      this.status = { ...this.status, ...status }
      this.controlEnabled = status.connected
    }
  }
})
```

告警等模块可按页面复杂度决定是否集中入 Pinia，或由视图本地状态 + WebSocket 维护。
